# 泛型

## 函数使用泛型

```js ts
// 如果无法完成推导，且未传递具体类型，默认为空对象
function take<T = number>(arr:T[],n:number):T[]{
    if(n >= arr.length){
        return arr;
    }
    const newArr = T[] = [];
    for(let i = 0; i < n; i ++){
        newArr.push(arr[i]);
    }
    return newArr[];
}
```

## 在类型别名 接口 类 中 使用泛型

直接在名称后写泛型

### 类型别名

```js ts
type callback<T> = (n: T, i: number) => boolean;

function filter<T>(arr :T[],callback: callback<T>):T[]{
    const newArr:T[]= [];
    arr.forEach(n=>{
        if(callback(n,i)){
            newArr.push(n);
        }
    })
    return newArr;
}

```

### 接口

```js ts
interface callback<T> { 
    (n: T, i: number) : boolean;
}

function filter<T>(arr :T[],callback: callback<T>):T[]{
    const newArr:T[]= [];
    arr.forEach(n=>{
        if(callback(n,i)){
            newArr.push(n);
        }
    })
    return newArr;
}

```

### 类

```js ts
export class ArrayHelper<T> {
    
    constructor(private arr:T[]){

    }


    take(n:number):T[]{
        if(n >= this.arr.length){
            return this.arr;
        }
        const newArr = T[] = [];
        for(let i = 0; i < n; i ++){
            newArr.push(this.arr[i]);
        }
        return newArr[];
    }

    private getRandom(min: number,max: number){
        const dec = max - min;
        return Math.floor(Math.random() * dec + max);
    }

    shuffle(){
        for(let i = 0; i< this.arr.length; i++){
            const targetIndex = this.getRandom(0, this.arr.length);
            const temp = this.arr[i];
            this.arr[i] = this.arr[targetIndex];
            this.arr[targetIndex] = temp;
        }
    }
}


```

## 泛型约束

```js ts
interface hasNameProperty<T> { 
    name: string
}

function nameToUpperCase<T extends hasNameProperty>(obj: T):T{
    obj.name.split(" ").map(s=>if(s.length>0) s[0].toUpperCase() + s.substr(1))
    return obj;
}

```

## 多泛型

```js ts
//[1,3,4] + ["a","b","c"]  = [1,3,4,"a","b","c"]
function mixinArray<T1,T2>(arr1: T1[],arr2: T2):(T1|T2)[]{
    if(arr1.length != arr2.length){
        throw new Error("两数组不相等");
    }
    let result:(T1|T2)[]=[];
    for(let i = 0; i < arr1.length; i++){
        result.push(arr1[i]);
        result.push(arr2[i]);
    }
    return result;
}

```
