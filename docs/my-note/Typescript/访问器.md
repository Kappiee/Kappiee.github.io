# 访问器

```js ts
class User{
    readonly id : number


    constructor(public name:string,private age:number){ 
        this.id = Math.random();
    }

    set age(value: number){
        if(value < 0){
            this._age = 0;
        }else if (value > 200){
            this._age = 200;
        }else{
            this._age = value
        }
    }

    get age(){
        return Math.floor(this._age;)
    }

}

```
