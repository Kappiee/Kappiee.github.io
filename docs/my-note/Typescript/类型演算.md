# 类型演算

## 三个关键字

### typeof

#### 用于类型约束时

TS中的typeof,书写位置在类型约束的位置上
表示获取某个数据的类型

```js ts
const a = "13123";
// 当不知道a的类型是什么时，使用typeof使b与a类型保持一致
let b :typeof a = "asd";
```

#### 用于类时

表示构造函数

```js ts
class User{
    loginId:string
    loginPwd:string
}

// 传入User类名 返回实体类
function createUser(cls: new () => User):User{
    return new cls();
}

const u = createUser(User);

// 使用typeof的方式
function createUser(cls: typeof User):User{
    return new cls();
}

// 另一个例子
const u = new User()
const u2 = u//u、u2为User类型

const A = User // A 的类型为 typeof User
// 此时u3也为User类型
const u3 = new A()

```

### keyof

作用于类、接口、类型别名,用于获取其他类型中的所有成员名组成成员名，租成的联合类型

```js ts
class User{
    loginId:string
    loginPwd:string
    age:number
}

// 报错
function printUserProperty(obj: User, prop: string){
    console.log(obj[prop]);
}

// 正常
function printUserProperty(obj: User, prop: "loginId"|"loginPwd"|"age" ){
    console.log(obj[prop]);
}

// keyof
function printUserProperty(obj: User, prop: keyof User ){
    console.log(obj[prop]);
}
const u: User = {
    loginId:"001",
    loginPwd:"13213123",
    age:132
}
printUserProperty(u,"age");

```

## in

该关键字往往和keyof联用，限制某个索引类型的取值范围

```js ts
interface User{
    loginId:string
    loginPwd:string
    age:number
}

// 索引器
type UserString = {
    [p:string]:string
}

const u: UserString = {};

u.abc = "123"

// 限制索引器
type UserString = {
    [p in "loginId"|"loginPwd"|"age"]:string
}
// 联合keyof
type UserString = {
    [p in keyof User]:string
}
// 进阶
type UserString = {
    [p in keyof User]:User[p]
}
type UserReadonly = {
   readonly [p in keyof User]:User[p]
}
type Readonly<T> = {
   readonly [p in keyof T]:T[p]
}
type String<T>= {
   [p in keyof T]:string
}
```
